# Recalbox's Pad-to-Keyboard configuration
#{GameID}
#DOS
 
# dpad to arrow keys
0:up = up
0:down = down
0:right = right
0:left = left
0:j1up = up
0:j1down = down
0:j1right = right
0:j1left = left

# Start c'est Enter
0:start = enter ;; Entree
# Select c'est Escape
0:select = esc ;; Retour

# X c'est ctrl
0:b = leftctrl ;; Fire
# O c'est '
0:a = S07 ;; Next Weapon
# /\ c'est espace
0:x = space
# [] c'est Alt
0:y = leftalt ;; Strafe

#L1 c'est PageUp
0:l1 = home ;; Aim up
#L2 c'est PageDown
0:l2 = end ;; Aim down

#R1 c'est Q
0:r1 = a ;; Jump
#R2 c'est C
0:r2 = z ;; Crouch
#L3 c'est N
0:l3 = n ;; No (N)
#R3 c'est Y
0:r3 = y ;; Yes (Y)

#hotkey c'est quitter
0:hotkey = F9 ;; Hotkey du Reset pour quitter
